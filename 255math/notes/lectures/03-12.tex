\begin{example}
  Let $f_n(x) = nx(1 - x)^n, \ x \in [0, 1]$. Then
  \begin{gather*}
    f_n(x) \to 0, \ \forall x \in [0, 1]
  \end{gather*} 
  Observation:
  \begin{gather*}
    \lim_{n \to \infty}n^kr^n = 0, \ \ \text{if} \ |r| < 1
  \end{gather*} 
\end{example} 

\begin{recall}
  $C(X) = \{f$ bounded and continuous on $X \}$.
  \begin{gather*}
    ||f||_\infty = \sum_{x \in X}|f(x)|
  \end{gather*} 
  The distance function on $C(X)$ is 
  \begin{gather*}
    d(f, g) = ||f - g||_\infty
  \end{gather*} 
\end{recall} 

\begin{corollary}
  $C(X)$ is a complete metric space. 

  \begin{proof}
    We need to show than any cauchy sequence $\{f_n\} \in C(X)$ converges to a bounded continuous function $f$. 
    \\\\
    Relationship with uniform convergence:. $\{f_n\}$ cauchy in $C(X) \Leftrightarrow \{f_n\}$ satisfies cauchy criterion $\Rightarrow f_n \to f$ uniformly $\Rightarrow f$ is continuous. % revisit
    \\\\
  \end{proof} 
\end{corollary}
